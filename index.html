// Fonction pour extraire l'identifiant depuis les paramètres de l'URL function getIdFromUrl() { const urlParams = new URLSearchParams(window.location.search); return urlParams.get("id"); // Récupère la valeur du paramètre 'id' } // Fonction pour envoyer l'identifiant au backend function verifyId() { const id = getIdFromUrl(); const resultDiv = document.getElementById("result"); if (!id) { resultDiv.textContent = "❌ Aucun identifiant fourni."; return; } fetch("/verify", { method: "POST", headers: { "Content-Type": "application/json", }, body: JSON.stringify({ id: id }), }) .then((response) => response.json()) .then((data) => { if (data.exists) { resultDiv.textContent = `✅ L'identifiant ${id} est valide : Existe dans la base de données.`; } else { resultDiv.textContent = `❌ L'identifiant ${id} est invalide : N'existe pas dans la base de données.`; } }) .catch((error) => { console.error("Erreur :", error); resultDiv.textContent = "❌ Erreur de connexion au serveur."; }); } // Appel de la fonction au chargement de la page verifyId();
